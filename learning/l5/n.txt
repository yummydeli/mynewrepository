#include <iostream>
#include <opencv2/opencv.hpp>
using namespace std;

int main() {
    using namespace cv;

    auto mat = imread(
            "test.jpg",
            cv::IMREAD_GRAYSCALE/*IMPORTANT IF THRESHOLDING*/
            );
    resize(mat,mat,{200,200});//zoom in
    imshow("test",mat);
    threshold(mat,mat,160,255,cv::THRESH_BINARY);
    imshow("test2",mat);
    waitKey();
    std::cout << "Hello, World!" << std::endl;
    return 0;
}
